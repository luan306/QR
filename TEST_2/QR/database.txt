CREATE DATABASE inventory;
USE inventory;

-- =========================
-- 1. BẢNG BỘ PHẬN
-- =========================
CREATE TABLE departments (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL
);

-- =========================
-- 2. BẢNG LOẠI THIẾT BỊ
-- =========================
CREATE TABLE device_types (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL
);

-- =========================
-- 3. BẢNG USER
-- =========================
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    full_name VARCHAR(100),
    role ENUM('admin','user') DEFAULT 'user',
    department_id INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (department_id) REFERENCES departments(id)
);

-- =========================
-- 4. BẢNG THIẾT BỊ
-- =========================
CREATE TABLE devices (
    id INT AUTO_INCREMENT PRIMARY KEY,
    qr_code VARCHAR(255) UNIQUE NOT NULL,
    name VARCHAR(100),
    location VARCHAR(100),

    device_type_id INT,
    original_department_id INT,
    current_department_id INT,

    status ENUM('active','broken','lost') DEFAULT 'active',

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (device_type_id) REFERENCES device_types(id),
    FOREIGN KEY (original_department_id) REFERENCES departments(id),
    FOREIGN KEY (current_department_id) REFERENCES departments(id)
);

-- =========================
-- 5. BẢNG LỊCH SỬ QUÉT
-- =========================
CREATE TABLE scans (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    device_id INT,

    scan_department_id INT,
    scanned_at DATETIME DEFAULT CURRENT_TIMESTAMP,

    note VARCHAR(255),

    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (device_id) REFERENCES devices(id),
    FOREIGN KEY (scan_department_id) REFERENCES departments(id)
);

SET @c = (SELECT COUNT(*) FROM information_schema.COLUMNS
  WHERE TABLE_SCHEMA='inventory' AND TABLE_NAME='devices' AND COLUMN_NAME='department_id');

CREATE TABLE IF NOT EXISTS audit_sessions (
  id            INT AUTO_INCREMENT PRIMARY KEY,
  user_id       INT NOT NULL,
  department_id INT NOT NULL,
  started_at    DATETIME DEFAULT CURRENT_TIMESTAMP,
  ended_at      DATETIME NULL,
  FOREIGN KEY (user_id)       REFERENCES users(id),
  FOREIGN KEY (department_id) REFERENCES departments(id)
);
